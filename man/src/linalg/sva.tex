\mansection{sva}
\begin{mandesc}
  \short{sva}{singular value approximation} \\ % 
\end{mandesc}
%\index{sva}\label{sva}
%-- Calling sequence section
\begin{calling_sequence}
\begin{verbatim}
  [U,s,V]=sva(A,k)  
  [U,s,V]=sva(A,tol)  
\end{verbatim}
\end{calling_sequence}
%-- Parameters
\begin{parameters}
  \begin{varlist}
    \vname{A}: real or complex matrix
    \vname{k}: integer
    \vname{tol}: nonnegative real number
  \end{varlist}
\end{parameters}
\begin{mandescription}
  Singular value approximation.\verb![U,S,V]=sva(A,k)! with \verb!k! an integer
  $>$=1, returns \verb!U,S! and \verb!V! such that
  \verb!B=U*S*V'! is the best L2 approximation of
  \verb!A! with rank(\verb!B!)=\verb!k!.\verb![U,S,V]=sva(A,tol)! with \verb!tol! a real
  number, returns \verb!U,S! and \verb!V! such that
  \verb!B=U*S*V'! such that L2-norm of \verb!A-B!
  is at most \verb!tol!.
\end{mandescription}
%--example 
\begin{examples}
  \begin{mintednsp}{nsp}
    A=rand(5,4)*rand(4,5);
    [U,s,V]=sva(A,2);
    B=U*s*V';
    svd(A)
    svd(B)
    clean(svd(A-B))
  \end{mintednsp}
\end{examples}
