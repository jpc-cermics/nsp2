\mansection{sylv}
\begin{mandesc}
  \short{sylv}{Sylvester equation} \\ % 
\end{mandesc}
%\index{sylv}\label{sylv}
%-- Calling sequence section
\begin{calling_sequence}
\begin{verbatim}
  sylv(A,B,C,flag)  
\end{verbatim}
\end{calling_sequence}
%-- Parameters
\begin{parameters}
  \begin{varlist}
    \vname{A,B,C}: three real matrices of appropriate dimensions.
    \vname{flag} character string (\verb!'c'! or \verb!'d'!)
  \end{varlist}
\end{parameters}
\begin{mandescription}
  \verb!X = sylv(A,B,C,'c')! computes \verb!X!, solution
  of the "continuous time" Sylvester equation
\begin{verbatim}
  A*X+X*B=C 
\end{verbatim}
\verb!X=sylv(A,B,C,'d')! computes \verb!X!, solution
of the "discrete time" Sylvester equation
\begin{verbatim}
  A*X*B-X=C
\end{verbatim}
If \verb![mA,nA]=size(A);[mB,nB]=size(B);[mC,nC]=size(C);!
\verb!mA! should be equal to \verb!mC! and \verb!nB!  should be equal to
\verb!nC!. 
Based on Lapack routines \verb!dtrsyl! and \verb!ztrsyl!.
\end{mandescription}
%--example 
\begin{examples}
  \begin{Verbatim}
    A=rand(4,4);C=rand(4,3);B=rand(3,3);
    X = sylv(A,B,C,'c');
    norm(A*X+X*B-C)
    X=sylv(A,B,C,'d') 
    norm(A*X*B-X-C)
  \end{Verbatim}
\end{examples}
