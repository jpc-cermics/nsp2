\mansection{cond}
\begin{mandesc}
  \short{cond}{condition number} \\ % 
\end{mandesc}
%\index{cond}\label{cond}
%-- Calling sequence section
\begin{calling_sequence}
\begin{verbatim}
  cond(X)  
\end{verbatim}
\end{calling_sequence}
%-- Parameters
\begin{parameters}
  \begin{varlist}
    \vname{X}: real or complex square matrix
  \end{varlist}
\end{parameters}
\begin{mandescription}
  Condition number in 2-norm.  \verb!cond(X)! is the  ratio  of  the
  largest singular value of  \verb!X!  to the smallest.
\end{mandescription}
%--example 
\begin{examples}
  \begin{mintednsp}{nsp}
    A=testmatrix('hilbert',6);
    cond(A)
  \end{mintednsp}
\end{examples}
