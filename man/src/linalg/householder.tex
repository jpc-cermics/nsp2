\mansection{householder}
\begin{mandesc}
  \short{householder}{Householder orthogonal reflexion matrix} \\ % 
\end{mandesc}
%\index{householder}\label{householder}
%-- Calling sequence section
\begin{calling_sequence}
\begin{verbatim}
  u=householder(v [,w])  
\end{verbatim}
\end{calling_sequence}
%-- Parameters
\begin{parameters}
  \begin{varlist}
    \vname{v}: real or complex column vector
    \vname{w}: real or complex column vector with same size as \verb!v!. 
    Default value is \verb!eye(size(v))!
    \vname{u}: real or complex column vector
  \end{varlist}
\end{parameters}
\begin{mandescription}
  given two column vectors \verb!v! and \verb! w! of same size,
  \verb!householder(v,w)! returns a unitary column vector \verb!u!, such that
  \verb! (eye()-2*u*u')*v! is proportional to \verb!w!.  The matrix
  \verb!eye(size(u*u'))-2*u*u'!  is the orthogonal Householder reflexion
  matrix. When vector \verb!w! is not given its value is assumed to be
  \verb! eye(v)! and the vector \verb!(eye(size(u*u'))-2*u*u')*v! is the vector
  \verb! eye(size(v))*norm(v)!.
\end{mandescription}

\begin{examples}
  \begin{mintednsp}{nsp}
    v=(1:5)';
    u=householder(v);
    if rank( [(eye(size(u*u'))-2*u*u')*v,eye(size(v))]) <> 1 then pause;end
  \end{mintednsp}
\end{examples}
