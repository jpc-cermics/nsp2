\mansection{sqroot}
\begin{mandesc}
  \short{sqroot}{W*W' hermitian factorization} \\ % 
\end{mandesc}
%\index{sqroot}\label{sqroot}
%-- Calling sequence section
\begin{calling_sequence}
\begin{verbatim}
  sqroot(X)  
\end{verbatim}
\end{calling_sequence}
%-- Parameters
\begin{parameters}
  \begin{varlist}
    \vname{X}: symmetric non negative definite real or complex matrix
  \end{varlist}
\end{parameters}
\begin{mandescription}
  returns W such that  \verb!X=W*W'! (uses SVD).
\end{mandescription}
%--example 
\begin{examples}
  \begin{mintednsp}{nsp}
    X=rand(5,2)*rand(2,5);X=X*X';
    W=sqroot(X)
    norm(W*W'-X,1)
    //
    X=rand(5,2)+%i*rand(5,2);X=X*X';
    W=sqroot(X)
    norm(W*W'-X,1)
  \end{mintednsp}
\end{examples}
