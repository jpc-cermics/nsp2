\mansection{coff}
\begin{mandesc}
  \short{coff}{resolvent (cofactor method)  } \\ % 
\end{mandesc}
%\index{coff}\label{coff}
%-- Calling sequence section
\begin{calling_sequence}
\begin{verbatim}
  [N,d]=coff(M [,var])   
\end{verbatim}
\end{calling_sequence}
%-- Parameters
\begin{parameters}
  \begin{varlist}
    \vname{M}: square real matrix
    \vname{var}: character string
    \vname{N}: polynomial matrix (same size as \verb!M!)
    \vname{d}: polynomial (characteristic polynomial \verb!poly(A,'s')!)
  \end{varlist}
\end{parameters}
\begin{mandescription}
  \verb!coff! computes R=\verb!(s*eye(size(M))-M)^-1! for a real matrix \verb!M!. 
  \verb!R! is given by \verb!N/d!. \verb!N! is the numerator as a polynomial matrix, \verb!d! is the 
  common denominator. \verb!var! character string ('\verb!s!' if omitted)
\end{mandescription}
%--example 
\begin{examples}
  \begin{mintednsp}{nsp}
    M=[1,2;0,3];
    [N,d]=coff(M,'s');
    s=poly(0,'s');
    P=N*(s*eye(size(M))-M) - d*eye(size(M));
    if norm(coeff(P)) > 100*%eps then pause;end
  \end{mintednsp}
\end{examples}
