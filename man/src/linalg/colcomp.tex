\mansection{colcomp}
\begin{mandesc}
  \short{colcomp}{column compression, kernel, nullspace} \\ % 
\end{mandesc}

%-- Calling sequence section
\begin{calling_sequence}
\begin{verbatim}
  [W,rk]=colcomp(A , meth=str, tol=rscal) 
\end{verbatim}
\end{calling_sequence}
% -- Parameters
\begin{parameters}
  \begin{varlist}
    \vname{A}: numerical matrix, real or complex.
    \vname{tol=rscal}: optional named argument, \verb+rscal+ must be  a real positive scalar (default \verb+sqrt(%eps)*norm(A,1)+).
    \vname{meth=str}: optional named argument, a string among \verb+"svd"+ (default) or \verb+"qr"+
    \vname{W}:  square non-singular matrix (change of basis)
    \vname{rk}: integer (rank of given matrix \verb!A!)
  \end{varlist}
\end{parameters}
\begin{mandescription}
  Column compression of matrix \verb!A!, that is \verb!Ac = A*W! is 
  column compressed i.e \verb!Ac=[0,Af]! with \verb!Af! full column rank, 
  rank(\verb!Af!) = rank(\verb!A!) = \verb!rk!.\verb!flag! and \verb!tol! are optional parameters: \verb!flag = 'qr'! 
  or \verb!'svd'! (default is \verb!'svd'!).\verb!tol! = tolerance parameter (of order \verb!%eps! as default value).
  The \verb!ma-rk! first columns of \verb!W! span the kernel of \verb!A! 
  when \verb!size(A)=(na,ma)!
\end{mandescription}
%--example 
\begin{examples}
  \begin{mintednsp}{nsp}
    A=rand(5,2)*rand(2,5);
    [X,r]=colcomp(A);
    norm(A*X(:,1:$-r),1)
  \end{mintednsp}
\end{examples}
