\mansection{spanplus}
\begin{mandesc}
  \short{spanplus}{sum of subspaces} \\ % 
\end{mandesc}
%\index{spanplus}\label{spanplus}
%-- Calling sequence section
\begin{calling_sequence}
\begin{verbatim}
  [X,dim,dima]=spanplus(A,B[,tol])   
\end{verbatim}
\end{calling_sequence}
%-- Parameters
\begin{parameters}
  \begin{varlist}
    \vname{A, B}: two real or complex matrices with equal number of rows
    \vname{X}: orthogonal or unitary square matrix
    \vname{dim, dima}: integers, dimension of subspaces
    \vname{tol}: nonnegative real number
  \end{varlist}
\end{parameters}
\begin{mandescription}
  computes a  basis X such that:
  the first \verb!dima! columns of \verb!X! span Range(\verb!A!)
  and the following (\verb!dim-dima!) columns make a basis of \verb!A+B!
  relative to \verb!A!.
  The \verb!dim! first columns of \verb!X! make a basis for \verb!A+B!.
  One has the following canonical form for \verb![A,B]!:
\begin{verbatim}
           [*,*]    (dima rows)
  X'*[A,B]=[0,*]    (dim-dima rows)
           [0,0]    
\end{verbatim}
\verb!tol! is an optional argument (see function code).
\end{mandescription}
%--example 
\begin{examples}
  \begin{mintednsp}{nsp}
    A=rand(6,2)*rand(2,5);      // rank(A)=2
    B=[A(:,1),rand(6,2)]*rand(3,3);   //two additional independent vectors
    [X,dim,dimA]=spanplus(A,B);
    if dimA<>2 then pause;end 
    if dim<>4 then pause;end 
  \end{mintednsp}
\end{examples}
