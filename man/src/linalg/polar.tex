\mansection{polar}
\begin{mandesc}
  \short{polar}{polar form} \\ % 
\end{mandesc}
%\index{polar}\label{polar}
%-- Calling sequence section
\begin{calling_sequence}
\begin{verbatim}
  [Ro,Theta]=polar(A)  
\end{verbatim}
\end{calling_sequence}
%-- Parameters
\begin{parameters}
  \begin{varlist}
    \vname{A}: real or complex square matrix
    \vname{Ro,  }: real matrix
    \vname{Theta,  }: real or complex matrix
  \end{varlist}
\end{parameters}
\begin{mandescription}
  \verb![Ro,Theta]=polar(A)! returns the polar form of
  \verb!A! i.e.  \verb!A=Ro*expm(%i*Theta)!\verb!Ro! symmetric $>$=0 and \verb!Theta! hermitian
  $>$=0.
\end{mandescription}
%--example 
\begin{examples}
  \begin{mintednsp}{nsp}
    A=rand(5,5);
    [Ro,Theta]=polar(A);
    norm(A-Ro*expm(%i*Theta),1)
  \end{mintednsp}
\end{examples}
