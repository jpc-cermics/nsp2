\mansection{givens}
\begin{mandesc}
  \short{givens}{Givens transformation} \\ % 
\end{mandesc}
%\index{givens}\label{givens}
%-- Calling sequence section
\begin{calling_sequence}
\begin{verbatim}
  U=givens(xy)  
  U=givens(x,y)  
  [U,c]=givens(xy)  
  [U,c]=givens(x,y)  
\end{verbatim}
\end{calling_sequence}
%-- Parameters
\begin{parameters}
  \begin{varlist}
    \vname{x,y}: two real or complex numbers
    \vname{xy}:  real or complex size 2 column vector
    \vname{U}:  a \verb!2!x\verb!2! unitary matrix
    \vname{c}:  real or complex size 2 column vector
  \end{varlist}
\end{parameters}
\begin{mandescription}
  \verb!U = givens(x, y)! or \verb!U = givens(xy)! with \verb!xy = [x;y]!
  returns a \verb!2!x\verb!2! unitary matrix \verb!U! such
  that:\verb!U*xy=[r;0]=c!.  Note that \verb!givens(x,y)! and
  \verb!givens([x;y])! are equivalent.
\end{mandescription}
%--example 
\begin{examples}
  \begin{mintednsp}{nsp}
    A=[3,4;5,6];
    U=givens(A(:,1));
    U*A
  \end{mintednsp}
\end{examples}
