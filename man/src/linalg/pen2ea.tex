\mansection{pen2ea}
\begin{mandesc}
  \short{pen2ea}{pencil to E,A conversion} \\ % 
\end{mandesc}
%\index{pen2ea}\label{pen2ea}
%-- Calling sequence section
\begin{calling_sequence}
\begin{verbatim}
  [E,A]=pen2ea(Fs)  
\end{verbatim}
\end{calling_sequence}
%-- Parameters
\begin{parameters}
  \begin{varlist}
    \vname{Fs}: matrix pencil \verb!s*E-A!
    \vname{E,A}: two matrices such that \verb!Fs=s*E-A!
  \end{varlist}
\end{parameters}
\begin{mandescription}
  Utility function. Given the pencil \verb!Fs=s*E-A!,
  returns the matrices \verb!E! and \verb!A!.
\end{mandescription}
%--example 
\begin{examples}
  \begin{mintednsp}{nsp}
    E=[1,0];A=[1,2];s=poly(0,'s');
    [E,A]=pen2ea(s*E-A)
  \end{mintednsp}
\end{examples}
