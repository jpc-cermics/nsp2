\mansection{factor}
\begin{mandesc}
   \short{factor}{prime factorization of an integer}
\end{mandesc}

%-- Calling sequence section
\begin{calling_sequence}
\begin{verbatim}
 f = factor(n)
 [f,p] = factor(n)
\end{verbatim}
\end{calling_sequence}

%-- Parameters
\begin{parameters}
  \begin{varlist}
   \vname{n}: positive integer (less or equal than 2^32)
   \vname{f}: row vector with the prime factorization of $n$.
   \vname{p}: if required a vector with the multiplicity of each factor.
  \end{varlist}
\end{parameters}

\begin{mandescription}
  Computes the prime factorization of an integer $n$. We must have $n
  \le 2^{32}$. With one output argument each individual prime factor
  of $n$ appears a number of time equal to its multiplicity. With two
  output arguments the multiplicity is given by the second output
  argument $p$, each factor appearing one time in the vector $f$.

\end{mandescription}

%--example 
\begin{examples}
\begin{mintednsp}{nspxxx}
p=factor(2^4*5^7)
[p,q]=factor(2^4*5^7)
\end{mintednsp}

\end{examples}
