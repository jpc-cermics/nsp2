\mansection{primes}
\begin{mandesc}
   \short{primes}{compute prime numbers}
\end{mandesc}

%-- Calling sequence section
\begin{calling_sequence}
\begin{verbatim}
 p = primes(n)
\end{verbatim}
\end{calling_sequence}

%-- Parameters
\begin{parameters}
  \begin{varlist}
   \vname{n} : integer (or real) scalar
   \vname{p} : row vector with primes numbers
  \end{varlist}
\end{parameters}

\begin{mandescription}
  Computes all prime numbers inferior or equal to $n$ with 
  Erathostene 's sieve.
\end{mandescription}

%--example 
\begin{examples}
\begin{program}\HCode{p=primes(20)\Hnewline
// p must be equal to [2,3,5,7,11,13,17,19]}
\end{program}
\end{examples}
