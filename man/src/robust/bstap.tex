\mansection{bstap}
\begin{mandesc}
  \short{bstap}{hankel approximant} \\ % 
\end{mandesc}
%\index{bstap}\label{bstap}
%-- Calling sequence section
\begin{calling_sequence}
\begin{verbatim}
  [Q]=bstap(Sl)  
\end{verbatim}
\end{calling_sequence}
%-- Parameters
\begin{parameters}
  \begin{varlist}
    \vname{sl}: linear system (\verb!syslin!) assumed to be continuous-time and anti-stable.
    \vname{Q}: best stable approximation of \verb!Sl! (\verb!syslin!).
  \end{varlist}
\end{parameters}
\begin{mandescription}
  Computes the best approximant \verb!Q! of the linear system \verb!Sl! where
  \verb!||T||! is the  $H$-infinity norm of the Hankel operator associated with \verb!Sl!.
\end{mandescription}
