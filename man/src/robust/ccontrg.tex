\mansection{ccontrg}
\begin{mandesc}
  \short{ccontrg}{central H-infinity controller} \\ % 
\end{mandesc}
%\index{ccontrg}\label{ccontrg}
%-- Calling sequence section
\begin{calling_sequence}
\begin{verbatim}
  [K]=ccontrg(P,r,gamma);  
\end{verbatim}
\end{calling_sequence}
%-- Parameters
\begin{parameters}
  \begin{varlist}
    \vname{P}: \verb!syslin! list (linear system in state-space representation)
    \vname{r}: 1x2 row vector, dimension of the 2,2 part of \verb!P!
    \vname{gamma}: real number
  \end{varlist}
\end{parameters}
\begin{mandescription}
  returns a realization \verb!K! of the central controller for the
  general standard problem in state-space form.
  Note that \verb!gamma! must be $>$ gopt (ouput of \verb!gamitg!).
  The linear system \verb!P! contains the parameters of plant realization \verb!(A,B,C,D)!
  with
\begin{verbatim}
  B = ( B1 , B2 ) ,        C= ( C1 ) ,    D = ( D11  D12)
                              ( C2 )          ( D21  D22)
\end{verbatim}
\verb!r(1)! and \verb!r(2)! are the dimensions of \verb!D22! (rows x columns)
\end{mandescription}
