\mansection{lcm}
\begin{mandesc}
  \short{lcm}{least common multiple} \\ % 
\end{mandesc}
%\index{lcm}\label{lcm}
%-- Calling sequence section
\begin{calling_sequence}
\begin{verbatim}
  [pp,fact]=lcm(p)  
\end{verbatim}
\end{calling_sequence}
%-- Parameters
\begin{parameters}
  \begin{varlist}
    \vname{p}:
    \vname{fact}: polynomial vector or integer vector (type equal to 8)
    \vname{pp}: polynomial or integer
  \end{varlist}
\end{parameters}
\begin{mandescription}
  \verb!pp=lcm(p)! computes the lcm \verb!pp! of polynomial vector \verb!p!.\verb![pp,fact]=lcm(p)! computes in addition the vector \verb!fact! such that:\verb!p.*fact=pp*ones(p)!
\end{mandescription}
%--example 
\begin{examples}
  \begin{itemize}
  \item polynomial case
    \begin{mintednsp}{nsp}
      s=poly(0,'s');
      p=[s,s*(s+1)^2,s^2*(s+2)];
      [pp,fact]=lcm(p);
      p.*fact, pp
    \end{mintednsp}
  \item integer case
    \begin{mintednsp}{nsp}
      V=m2i([2^2*3^5, 2^3*3^2,2^2*3^4*5]);
      lcm(V)
    \end{mintednsp}
  \item scalar case
    \begin{mintednsp}{nsp}
      V=m2i([2^2*3^5, 2^3*3^2,2^2*3^4*5]);
      lcm(V)
    \end{mintednsp}
  \end{itemize}
\end{examples}
