\mansection{detr}
\begin{mandesc}
  \short{detr}{polynomial determinant} \\ % 
\end{mandesc}
%\index{detr}\label{detr}
%-- Calling sequence section
\begin{calling_sequence}
\begin{verbatim}
  d=detr(h)  
\end{verbatim}
\end{calling_sequence}
%-- Parameters
\begin{parameters}
  \begin{varlist}
    \vname{h}: polynomial or rational square matrix
  \end{varlist}
\end{parameters}
\begin{mandescription}
  \verb!d=detr(h)! returns the determinant \verb!d! of the polynomial or 
  rational function matrix \verb!h!. Based on Leverrier's algorithm.
\end{mandescription}
