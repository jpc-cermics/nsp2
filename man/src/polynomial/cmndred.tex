\mansection{cmndred}
\begin{mandesc}
  \short{cmndred}{common denominator form} \\ % 
\end{mandesc}
%\index{cmndred}\label{cmndred}
%-- Calling sequence section
\begin{calling_sequence}
\begin{verbatim}
  [n,d]=cmndred(num,den)  
\end{verbatim}
\end{calling_sequence}
%-- Parameters
\begin{parameters}
  \begin{varlist}
    \vname{num, den}: two polynomial matrices of same dimensions
  \end{varlist}
\end{parameters}
\begin{mandescription}
  \verb![n,d]=cmndred(num,den)! computes a polynomial matrix
  \verb!n! and a common denominator polynomial \verb!d!
  such that:\verb!n/d=num./den!
  The rational matrix defined by \verb!num./den! is \verb!n/d!
\end{mandescription}
