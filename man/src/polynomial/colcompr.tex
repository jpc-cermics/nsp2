\mansection{colcompr}
\begin{mandesc}
  \short{colcompr}{column compression of polynomial matrix} \\ % 
\end{mandesc}
%\index{colcompr}\label{colcompr}
%-- Calling sequence section
\begin{calling_sequence}
\begin{verbatim}
  [Y,rk,ac]=colcompr(A);  
\end{verbatim}
\end{calling_sequence}
%-- Parameters
\begin{parameters}
  \begin{varlist}
    \vname{A}: polynomial matrix
    \vname{Y}: square polynomial matrix (right unimodular basis)
    \vname{rk}: normal rank of \verb!A!
    \vname{Ac}: \verb!Ac=A*Y!, polynomial matrix
  \end{varlist}
\end{parameters}
\begin{mandescription}
  column compression of polynomial matrix \verb!A! (compression to the left)
\end{mandescription}
%--example 
\begin{examples}
  \begin{mintednsp}{nsp}
    s=poly(0,'s');
    p=[s;s*(s+1)^2;2*s^2+s^3];
    [Y,rk,ac]=colcompr(p*p');
    p*p'*Y
  \end{mintednsp}
\end{examples}
