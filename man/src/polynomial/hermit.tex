\mansection{hermit}
\begin{mandesc}
  \short{hermit}{Hermite form} \\ % 
\end{mandesc}
%\index{hermit}\label{hermit}
%-- Calling sequence section
\begin{calling_sequence}
\begin{verbatim}
  [Ar,U]=hermit(A)  
\end{verbatim}
\end{calling_sequence}
%-- Parameters
\begin{parameters}
  \begin{varlist}
    \vname{A}: polynomial matrix
    \vname{Ar}: triangular polynomial matrix
    \vname{U}: unimodolar polynomial matrix
  \end{varlist}
\end{parameters}
\begin{mandescription}
  Hermite form: \verb!U! is an unimodular matrix such that \verb!A*U! is
  in Hermite triangular form:
  The output variable is \verb!Ar=A*U!.
  Warning: Experimental version
\end{mandescription}
%--example 
\begin{examples}
  \begin{mintednsp}{nsp}
    s=poly(0,'s');
    p=[s, s*(s+1)^2, 2*s^2+s^3];
    [Ar,U]=hermit(p'*p);
    clean(p'*p*U), det(U)
  \end{mintednsp}
\end{examples}
