\mansection{systmat}
\begin{mandesc}
  \short{systmat}{system matrix} \\ % 
\end{mandesc}
%\index{systmat}\label{systmat}
%-- Calling sequence section
\begin{calling_sequence}
\begin{verbatim}
  [Sm]=systmat(Sl);  
\end{verbatim}
\end{calling_sequence}
%-- Parameters
\begin{parameters}
  \begin{varlist}
    \vname{Sl}: linear system (\verb!syslin! list) or descriptor system
    \vname{Sm}: matrix pencil
  \end{varlist}
\end{parameters}
\begin{mandescription}
  System matrix of the linear system \verb!Sl! (\verb!syslin! list) 
  in state-space form (utility function).
  \begin{equation*}
    Sm = \left( \begin{array}{cl} -s I + A & B \\C & D  \end{array}  \right)
  \end{equation*}
  For a descriptor system (\verb!Sl=list('des',A,B,C,D,E)!), \verb!systmat! returns:
  \begin{equation*}
    Sm = \left( \begin{array}{cl} -s E + A & B \\  C & D   \end{array} \right)
  \end{equation*}
\end{mandescription}
