\mansection{polfact}
\begin{mandesc}
  \short{polfact}{minimal factors} \\ % 
\end{mandesc}
%\index{polfact}\label{polfact}
%-- Calling sequence section
\begin{calling_sequence}
\begin{verbatim}
  [f]=polfact(p)  
\end{verbatim}
\end{calling_sequence}
%-- Parameters
\begin{parameters}
  \begin{varlist}
    \vname{p}: polynomial
    \vname{f}: vector \verb![f0 f1 ... fn]! such that \verb!p=prod(f)!
    \begin{varlist}
      \vname{f0}: constant
      \vname{fi}: polynomial
    \end{varlist}
  \end{varlist}
\end{parameters}
\begin{mandescription}
  \verb!f=polfact(p)! returns the minimal factors of p i.e. \verb!f=[f0 f1 ... fn]! such 
  that \verb!p=prod(f)!
\end{mandescription}
