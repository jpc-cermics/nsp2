\mansection{iirlp}
\begin{mandesc}
  \short{iirlp}{Lp IIR filter optimization} \\ % 
\end{mandesc}
%\index{iirlp}\label{iirlp}
%-- Calling sequence section
\begin{calling_sequence}
\begin{verbatim}
  [cost,grad,ind]=iirlp(x,ind,p,[flag],lambda,omega,ad,wa,td,wt)  
\end{verbatim}
\end{calling_sequence}
%-- Parameters
\begin{parameters}
  \begin{varlist}
    \vname{x}: 1X2 vector of the module and argument of the poles and the zeros of the filters
    \vname{flag}: string: \verb!'a'!  for amplitude, 'gd' for group delay; default case for amplitude and group delay.
    \vname{omega}: frequencies where the filter specifications are given
    \vname{wa,wt}: weighting functions for the amplitude and the group delay
    \vname{lambda}: weighting (with \verb!1-lambda!) of the costs (\verb!'a'! and \verb!'gd'! for getting the global cost.
    \vname{ad, td}: desired amplitude and group delay
    \vname{cost, grad}: criterium and gradient values
  \end{varlist}
\end{parameters}
\begin{mandescription}
  optimization of IIR filters for the Lp criterium for the
  amplitude and/or the group delay. (Rabiner \&amp; Gold pp270-273).
\end{mandescription}
