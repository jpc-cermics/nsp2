\mansection{find\_freq}
\begin{mandesc}
  \shortunder{find\_freq}{find_freq}{parameter compatibility for elliptic filter design} \\ % 
\end{mandesc}
%\index{find\_freq}\label{find-freq}
%-- Calling sequence section
\begin{calling_sequence}
\begin{verbatim}
  [m]=find_freq(epsilon,A,n)  
\end{verbatim}
\end{calling_sequence}
%-- Parameters
\begin{parameters}
  \begin{varlist}
    \vname{epsilon}: passband ripple
    \vname{A}: stopband attenuation
    \vname{n}: filter order
    \vname{m}: frequency needed for construction of elliptic filter
  \end{varlist}
\end{parameters}
\begin{mandescription}
  Search for m such that \verb!n=K(1-m1)K(m)/(K(m1)K(1-m))!
  with\verb!m1=(epsilon*epsilon)/(A*A-1)!;
  If \verb!m = omegar^2/omegac^2!, the parameters
  \verb!epsilon,A,omegac,omegar! and \verb!n! are then
  compatible for defining a prototype elliptic filter.
  Here, \verb!K=%k(m)! is the complete elliptic integral with parameter \verb!m!.
\end{mandescription}
