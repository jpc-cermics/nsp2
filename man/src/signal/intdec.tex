\mansection{intdec}
\begin{mandesc}
  \short{intdec}{Changes  sampling rate of a signal} \\ % 
\end{mandesc}
%\index{intdec}\label{intdec}
%-- Calling sequence section
\begin{calling_sequence}
\begin{verbatim}
  [y]=intdec(x,lom)  
\end{verbatim}
\end{calling_sequence}
%-- Parameters
\begin{parameters}
  \begin{varlist}
    \vname{x}: input sampled signal
    \vname{lom}: For a 1D signal this is a scalar which gives the rate
    change. For a 2D signal this is a 2-Vector of sampling rate changes
    \verb!lom!=(col rate change,row rate change) 
    \vname{y}: Output sampled signal
  \end{varlist}
\end{parameters}
\begin{mandescription}
  Changes the sampling rate of a 1D or 2D signal by the rates in lom
\end{mandescription}
