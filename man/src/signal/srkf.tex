\mansection{srkf}
\begin{mandesc}
  \short{srkf}{square root Kalman filter} \\ % 
\end{mandesc}
%\index{srkf}\label{srkf}
%-- Calling sequence section
\begin{calling_sequence}
\begin{verbatim}
  [x1,p1]=srkf(y,x0,p0,f,h,q,r)  
\end{verbatim}
\end{calling_sequence}
%-- Parameters
\begin{parameters}
  \begin{varlist}
    \vname{f, h}: current system matrices
    \vname{q, r}: covariance matrices of dynamics and observation noise
    \vname{x0, p0}: state estimate and error variance at t=0 based on data up to t=-1
    \vname{y}: current observation Output from the function is
    \vname{x1, p1}: updated estimate and error covariance at t=1 based on data up to t=0
  \end{varlist}
\end{parameters}
\begin{mandescription}
  square root Kalman filter algorithm
\end{mandescription}
