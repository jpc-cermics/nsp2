%-- Calling sequence section
\begin{calling_sequence}
\begin{verbatim}
  [m]=findm(chi)  
\end{verbatim}
\end{calling_sequence}
\begin{mandescription}
  Search for \verb!m! such that \verb!chi = %k(1-m)/%k(m)!
  (For use with \verb!find_freq!).
\end{mandescription}
