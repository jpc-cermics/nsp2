\mansection{remez}
\begin{mandesc}
  \short{remez}{Remez's algorithm} \\ % 
\end{mandesc}
%\index{remez}\label{remez}
%-- Calling sequence section
\begin{calling_sequence}
\begin{verbatim}
  [an]=remez(nc,fg,ds,wt)  
\end{verbatim}
\end{calling_sequence}
%-- Parameters
\begin{parameters}
  \begin{varlist}
    \vname{nc}: integer, number of cosine functions
    \vname{fg,ds,wt}: real vectors
    \vname{fg}: grid of frequency points in [0,.5)
      \vname{ds}: desired magnitude on grid \verb!fg!
      \vname{wt}: weighting function on error on grid \verb!fg!
  \end{varlist}
\end{parameters}
\begin{mandescription}
  minimax approximation of a frequency domain magnitude response.
  The approximation takes the form
\begin{verbatim}
  h = sum[a(n)*cos(wn)]
\end{verbatim}
for n=0,1,...,nc. An FIR, linear-phase filter
can be obtained from the the output of \verb!remez! by using the
following commands:
\begin{verbatim}
  hn(1nc-1)=an(nc-12)/2;
  hn(nc)=an(1);
  hn(nc+12*nc-1)=an(2nc)/2;
\end{verbatim}
where \verb!an! = cosine filter coefficients
\end{mandescription}
