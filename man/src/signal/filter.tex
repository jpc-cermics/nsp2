\mansection{filter}
\begin{mandesc}
  \short{filter}{modelling filter} \\ % 
\end{mandesc}
%\index{filter}\label{filter}
%-- Calling sequence section
\begin{calling_sequence}
\begin{verbatim}
  [y,xt]=filter(n,F,H,Rt,T)  
\end{verbatim}
\end{calling_sequence}
%-- Parameters
\begin{parameters}
  \begin{varlist}
    \vname{n}: number of computed points.
    \vname{F, H}: relevant matrices of the Markovian model.
    \vname{Rt, T}: gain matrices.
    \vname{y}: output of the filter.
    \vname{xt}: filter process.
  \end{varlist}
\end{parameters}
\begin{mandescription}
  This function computes the modelling filter
\end{mandescription}
