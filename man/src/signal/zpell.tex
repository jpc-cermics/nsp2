\mansection{zpell}
\begin{mandesc} 
  \short{zpell}{lowpass elliptic filter} \\ % 
\end{mandesc}
%\index{zpell}\label{zpell}
%-- Calling sequence section
\begin{calling_sequence}
\begin{verbatim}
  [zeros,poles,gain]=zpell(epsilon,A,omegac,omegar)  
\end{verbatim}
\end{calling_sequence}
%-- Parameters
\begin{parameters}
  \begin{varlist}
    \vname{epsilon}: real: ripple of filter in pass band (\verb!0$<$epsilon$<$1!)
    \vname{A}: real: attenuation of filter in stop band (\verb!A$>$1!)
    \vname{omegac}: real: pass band cut-off frequency in Hertz
    \vname{omegar}: real: stop band cut-off frequency in Hertz
    \vname{zeros}: resulting zeros of filter
    \vname{poles}: resulting poles of filter
    \vname{gain}: resulting gain of filter
  \end{varlist}
\end{parameters}
\begin{mandescription}
  Poles and zeros of prototype lowpass elliptic filter.
  \verb!gain! is the gain of the filter
\end{mandescription}
