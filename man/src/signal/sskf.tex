\mansection{sskf}
\begin{mandesc}
  \short{sskf}{steady-state Kalman filter} \\ % 
\end{mandesc}
%\index{sskf}\label{sskf}
%-- Calling sequence section
\begin{calling_sequence}
\begin{verbatim}
  [xe,pe]=sskf(y,f,h,q,r,x0)  
\end{verbatim}
\end{calling_sequence}
%-- Parameters
\begin{parameters}
  \begin{varlist}
    \vname{y}: data in form \verb![y0,y1,...,yn]!, \verb!yk! a column vector
    \vname{f}: system matrix dim(NxN)
    \vname{h}: observations matrix dim(MxN)
    \vname{q}: dynamics noise matrix dim(NxN)
    \vname{r}: observations noise matrix dim(MxM)
    \vname{x0}: initial state estimate
    \vname{xe}: estimated state
    \vname{pe}: steady-state error covariance
  \end{varlist}
\end{parameters}
\begin{mandescription}
  steady-state Kalman filter
\end{mandescription}
