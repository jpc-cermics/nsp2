\mansection{frmag}
\begin{mandesc}
  \short{frmag}{magnitude of FIR and IIR filters} \\ % 
\end{mandesc}
%\index{frmag}\label{frmag}
%-- Calling sequence section
\begin{calling_sequence}
\begin{verbatim}
  [xm,fr]=frmag(num[,den],npts)  
\end{verbatim}
\end{calling_sequence}
%-- Parameters
\begin{parameters}
  \begin{varlist}
    \vname{npts}: integer (number of points in frequency response)
    \vname{xm}: mvector of magnitude of frequency response at the points \verb!fr!
    \vname{fr}: points in the frequency domain where magnitude is evaluated
    \vname{num}: if \verb!den! is omitted vector coefficients/polynomial/rational polynomial of filter
    \vname{num}: if \verb!den! is given vector coefficients/polynomial of filter numerator
    \vname{den}: vector coefficients/polynomial of filter denominator
  \end{varlist}
\end{parameters}
\begin{mandescription}
  calculates the magnitude of the frequency responses of
  FIR and IIR filters. The filter description can be
  one or two vectors of coefficients, one or two polynomials,
  or a rational polynomial.
\end{mandescription}
