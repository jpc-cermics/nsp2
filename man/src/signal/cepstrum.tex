\mansection{cepstrum}
\begin{mandesc}
  \short{cepstrum}{cepstrum calculation} \\ % 
\end{mandesc}
%\index{cepstrum}\label{cepstrum}
%-- Calling sequence section
\begin{calling_sequence}
\begin{verbatim}
  fresp = cepstrum(w,mag)  
\end{verbatim}
\end{calling_sequence}
%-- Parameters
\begin{parameters}
  \begin{varlist}
    \vname{w}: positive real vector of frequencies (rad/sec)
    \vname{mag}: real vector of magnitudes (same size as \verb!w!)
    \vname{fresp}: complex vector
  \end{varlist}
\end{parameters}
\begin{mandescription}
  \verb!fresp = cepstrum(w,mag)! returns a frequency response \verb!fresp(i)!
  whose magnitude at frequency \verb!w(i)! equals \verb!mag(i)!  and such
  that the phase of \verb!freq! corresponds to a stable and minimum phase
  system. \verb!w! needs not to be sorted, but  minimal entry should not be
  close to zero and all the entries of \verb!w! should be different.
\end{mandescription}
%--example 
\begin{examples}
  \begin{mintednsp}{nsp}
    w=0.1:0.1:5;mag=1+abs(sin(w));
    fresp=cepstrum(w,mag);
    xclear();
    plot2d([w',w'],[mag(:),abs(fresp)])
  \end{mintednsp}
\end{examples}
