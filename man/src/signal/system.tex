\mansection{system}
\begin{mandesc}
  \short{system}{observation update} \\ % 
\end{mandesc}
%\index{system}\label{system}
%-- Calling sequence section
\begin{calling_sequence}
\begin{verbatim}
  [x1,y]=system(x0,f,g,h,q,r)  
\end{verbatim}
\end{calling_sequence}
%-- Parameters
\begin{parameters}
  \begin{varlist}
    \vname{x0}: input state vector
    \vname{f}: system matrix
    \vname{g}: input matrix
    \vname{h}: Output matrix
    \vname{q}: input noise covariance matrix
    \vname{r}: output noise covariance matrix
    \vname{x1}: output state vector
    \vname{y}: output observation
  \end{varlist}
\end{parameters}
\begin{mandescription}
  define system function which generates the next
  observation given the old state.
  System recursively calculated
\begin{verbatim}
  x1=f*x0+g*u
  y=h*x0+v
\end{verbatim}
where \verb!u! is distributed \verb!N(0,q)!
and \verb!v! is distribute \verb!N(0,r)!.
\end{mandescription}
