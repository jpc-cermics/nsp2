\mansection{sincd}
\begin{mandesc}
  \short{sincd}{digital sinc function or Direchlet kernel} \\ % 
\end{mandesc}
%\index{sincd}\label{sincd}
%-- Calling sequence section
\begin{calling_sequence}
\begin{verbatim}
  [s]=sincd(n,flag)  
\end{verbatim}
\end{calling_sequence}
%-- Parameters
\begin{parameters}
  \begin{varlist}
    \vname{n}: integer
    \vname{flag}: if \verb!flag = 1! the function is centred around the origin; if \verb!flag = 2! the function is delayed by \verb!%pi/(2*n)!
    \vname{s}: vector of values of the function on a dense grid of frequencies
  \end{varlist}
\end{parameters}
\begin{mandescription}
  function which calculates the function \verb!Sin(N*x)/N*Sin(x)!
\end{mandescription}
%--example 
\begin{examples}
  \begin{mintednsp}{nsp}
    plot2d([],sincd(10,1)) 
  \end{mintednsp}
\end{examples}
