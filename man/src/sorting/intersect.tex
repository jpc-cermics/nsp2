% -*- mode: latex -*-

\mansection{intersect}
\begin{mandesc}
  \short{intersect}{compute the set intersection of two vectors or two lists}
\end{mandesc}

% -- Calling sequence section
\begin{calling_sequence}
\begin{verbatim}
C = intersect(A,B)
[C, kA, kB] = intersect(A,B)
\end{verbatim}
\end{calling_sequence}
% -- Parameters
\begin{parameters}
  \begin{varlist}
    \vname{A}: vector of numbers, strings or cells (if A is a matrix, it is
    considered as a big column vector), or list.
    \vname{B}: vector of same type than A, or list if A is a list. 
    \vname{C}: vector or list with elements both in  $A$ and $B$.
    \vname{kA, kB}: vectors of indices such that $C = A(kA)$ and $C =
    B(kB)$ (equalities up to an increasing sort of $A(kA)$ and $B(kB)$
    in case of vectors of numbers or strings). 
  \end{varlist}
\end{parameters}

\begin{mandescription}
  This function computes the intersection $C = A \cap B$
  between 2 vectors (of numbers, strings or cells) or between 2
  lists, considered as sets. For vectors of numbers or strings
  $C$ is sorted in ascending order.

  $C$, $kA$ and $kB$ are row vectors if both A and B are row vectors, otherwise they
  are column vectors.
  
\end{mandescription}

\begin{examples}

\paragraph{example 1} with vectors of numbers:
\begin{program}\HCode{A = [0.5, -1, 2, 1, 2, -1, 0.5, 2, -1, 2];\Hnewline
B = [2, 0.5, 3, 912];\Hnewline
C = intersect(A,B)\Hnewline
[C, kA, kB] = intersect(A,B)\Hnewline
// we must have C = A(kA) and C = B(kB) up to a permutation\Hnewline
C.equal[sort(A(kA),"g","i")]\Hnewline
C.equal[sort(B(kB),"g","i")]}
\end{program}

\paragraph{example 2} with vectors of strings:
  \begin{program}\HCode{A = ["toto", "foo", "bar", "tumbi", "foobar", "bar", "toto", "foo", "bar"]\Hnewline
B = [ "foo", "bar", "nsp", "tumbi" ]\Hnewline
C = intersect(A,B)\Hnewline
[C, kA, kB] = intersect(A,B)\Hnewline
// we must have C = A(kA) and C = B(kB) up to a permutation\Hnewline
C.equal[sort(A(kA),"g","i")]\Hnewline
C.equal[sort(B(kB),"g","i")]}
  \end{program}

\paragraph{example 3} with vectors of cells:
  \begin{program}\HCode{A = {"toto", [0,1;1,0], "foo", 1, 2, 1, "bar"}\Hnewline
      B = {  "tumbi", 5 ,"foo", 2,  [0,1;1,0] }\Hnewline
      C = intersect(A,B)\Hnewline
      [C, kA, kB] = intersect(A,B)\Hnewline
      // we must have C = A(kA) and C = B(kB)\Hnewline
      C.equal[A(kA)]\Hnewline
      C.equal[B(kB)]}
 
  \end{program}

\paragraph{example 4} with lists:
  \begin{program}\HCode{A = list("toto", [0,1;1,0], "foo", 1, 2, 1, "bar")\Hnewline
    B = list(  "tumbi", 5 ,"foo", 2,  [0,1;1,0] )\Hnewline
    C = intersect(A,B)\Hnewline
    [C, kA, kB] = intersect(A,B)\Hnewline
    // we must have C = A(kA) and C = B(kB)\Hnewline
    C.equal[A.sublist[kA]]\Hnewline
    C.equal[B.sublist[kB]]}
  \end{program}

\end{examples}
