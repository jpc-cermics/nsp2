% -*- mode: latex -*-
\mansection{file}
\begin{mandesc}
  \short{file}{file and pathnames}
\end{mandesc}
%\index{file}\label{file}

%-- Calling sequence section
\begin{calling_sequence}
  \begin{verbatim}
    file(operation,S);
  \end{verbatim}
\end{calling_sequence}
%-- Parameters
\begin{parameters}
  \begin{varlist}
    \vname{operation}: a string giving an operation on filename
    \vname{S}: a string matrix.
  \end{varlist}
\end{parameters}
\begin{mandescription}
  The \verb!file! command is used to get file attributes, or 
  to get pathname informations, or to build pathnames from 
  components.
\end{mandescription}
%--example 
\begin{examples}
  \begin{program}\HCode{file("atime","doc.txt") ;\Hnewline
    file("attributes",file [,attrname, attrset]);\Hnewline
    file("attributes","doc.txt");\Hnewline
    //file("copy",["src","dst"])	    dst is a file \Hnewline
    //or \Hnewline
    //file("copy",["src1","src2","dst"])  dst is a directory \Hnewline
    file("mkdir","poo");\Hnewline
    file("copy",["doc.txt","doc1.txt"]) \Hnewline
    file("copy",["doc.txt","poo"]) \Hnewline
    // file("delete", "file" or "dir")     dir is deleted if it is empty \Hnewline
    file("delete","poo/doc.txt");\Hnewline
    file("delete","poo");\Hnewline
    path=file('join',[getcwd(),'doc.txt'])\Hnewline
    file("dirname",path)\Hnewline
    file("executable", path)\Hnewline
    file("exists",path)\Hnewline
    file("extension",path)\Hnewline
    file("isdirectory",path)\Hnewline
    file("isfile",path) \Hnewline
    file("lstat", path) \Hnewline
    file("mtime", path) \Hnewline
    file("nativename",path)\Hnewline
    file("owned", path)\Hnewline
    file("pathtype",path) \Hnewline
    file("readable",path) \Hnewline
    system("ln -f -s doc.txt doc.ln")\Hnewline
    if file("readlink","doc.ln")<>"doc.txt" then pause;end\Hnewline
    // file("rename",["src","dst"])\Hnewline
    // or \Hnewline
    // file("rename",["src1","src2","dst"])\Hnewline
    file("rename",["doc1.txt","doc2.txt"]);\Hnewline
    file("delete","doc2.txt");\Hnewline
    file("rootname", path)\Hnewline
    file("size", path)\Hnewline
    file("split", path)\Hnewline
    file("stat", path)\Hnewline
    file("tail", path)\Hnewline
    file("type", path)\Hnewline
    file("volumes")\Hnewline
    file("writable",path)\Hnewline
}
  \end{program}
\end{examples}
