\mansection{zeropen}
\begin{mandesc}
  \short{zeropen}{zero pencil} \\ % 
\end{mandesc}
%\index{zeropen}\label{zeropen}
%-- Calling sequence section
\begin{calling_sequence}
\begin{verbatim}
  [Z,U]=zeropen(Sl)  
\end{verbatim}
\end{calling_sequence}
%-- Parameters
\begin{parameters}
  \begin{varlist}
    \vname{Sl}: a linear system (\verb!syslin! list in state-space form \verb![A,B,C,D]!)
    \vname{Z}: matrix pencil \verb!Z=s*E-A!
    \vname{U}: square orthogonal matrix
  \end{varlist}
\end{parameters}
\begin{mandescription}
  \verb!Z = sE - F!  is the zero pencil of the linear system \verb!Sl! with 
  matrices \verb![A,B,C,D]!. Utility function.
  With \verb!U! row compression of \verb![B;D]! i.e, \verb!U*[B;D]=[0;*]!; one has:
\begin{verbatim}
  U*[-sI+A |B;   [ Z |0;      
    [   C  |D] = [ * |*]
\end{verbatim}
The zeros of \verb!Z! are the zeros of \verb!Sl!.
\end{mandescription}
