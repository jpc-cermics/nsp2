\mansection{acquire}
\begin{mandesc}
  \short{acquire}{search a variable in callers} \\
\end{mandesc}

% -- Calling sequence section
\begin{calling_sequence}
\begin{verbatim}
y = acquire(name);
\end{verbatim}
\end{calling_sequence}
% -- Parameters
\begin{parameters}
  \begin{varlist}
    \vname{name}: a string giving the name of symbol to search.
    \vname{y}: a nsp object.
  \end{varlist}
\end{parameters}

\begin{mandescription}
  This function is used to copy in the current frame a variable found in callers 
  with the given name. 
\end{mandescription}

\begin{examples}
\begin{Verbatim}
x=1:3;
function y=f(n)
   if n >= 10 then y=acquire('x'); 
   else y=f(n+1);
   end 
endfunction 
y=f(0);
\end{Verbatim}
\end{examples}
