% -*- mode: latex -*-

\mansection{m2s}
\begin{mandesc}
  \short{m2s}{convert to string matrix} \\
  \short{string}{convert object to string matrix}
\end{mandesc}
% -- Calling sequence section
\begin{calling_sequence}
\begin{verbatim}
S=string(Obj)
S=m2s(A [, format])
\end{verbatim}
\end{calling_sequence}

% -- Parameters
\begin{parameters}
  \begin{varlist}
    \vname{S}: a string matrix
    \vname{format}: a string 
    \vname{Obj}: an object 
    \vname{A}: a numerical matrix
  \end{varlist}
\end{parameters}
\begin{mandescription}
\verb+m2s+ is a conversion function from numerical matrices to string. 
It uses a default format for conversion which is similar to the default 
format used for displaying \verb+A+.
\verb+string+ emulates the Scilab \verb+string+ function.
\end{mandescription}
