% -*- mode: latex -*-

\mansection{part}
\begin{mandesc}
  \short{part}{substring extraction from indices}
\end{mandesc}
% -- Calling sequence section
\begin{calling_sequence}
\begin{verbatim}
 S2=part(S1,Ind,mode="bytes"|"utf8")
\end{verbatim}
\end{calling_sequence}
% -- Parameters
\begin{parameters}
  \begin{varlist}
    \vname{S1,S2}: string matrices.
    \vname{Ind}: a numerical matrix used as int vector (\verb+int(Ind(:))+).
    \vname{mode}: an optional string argument which can be \verb!'bytes'! or \verb!'utf8'!
    (default value is \verb!'bytes'!).
  \end{varlist}
\end{parameters}
\begin{mandescription}
  \verb!S2=part(S1,Ind)! returns a string matrix of same size as \verb+S1+.
  The entry \verb!S2(i,j)! is built from the entry \verb!S1(i,j)! as follows~:
\begin{verbatim}
  S2(i,j) = S1(i,j)[Ind(1)] + ... + S1(i,j)[Ind(size(Ind,'*'))]
\end{verbatim}
  where \verb!s[k]! stands for the one-character string made of the \verb!k!-th
  character of \verb!s! or made of a white space character if \verb!k! is outside the range
  of string \verb+s+.

  The \verb!k!-th character of a string is counted in bytes or in utf8 characters accoding
  to the \verb!mode! value.
\end{mandescription}
% --example
\begin{examples}
  \begin{mintednsp}{nsp}
    c=part(['a','abc','abcd'],[1,1,2]);
    S=part('cycloïde',[6,6,6]);
    \end{mintednsp}
\end{examples}
