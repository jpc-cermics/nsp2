% -*- mode: latex -*-

\mansection{strsubst}
\begin{mandesc}
  \short{strsubst}{sub-string substitution in a string matrix}
\end{mandesc}
% -- Calling sequence section
\begin{calling_sequence}
\begin{verbatim}
B=strsubst(A,str,rep)
\end{verbatim}
\end{calling_sequence}

% -- Parameters
\begin{parameters}
  \begin{varlist}
    \vname{B,A}: string matrices.
    \vname{str,rep}: strings.
  \end{varlist}
\end{parameters}

\begin{mandescription}
Replaces occurences of \verb+str+ by \verb+rep+ in string matrix \verb+A+.
\end{mandescription}
