% -*- mode: latex -*-

\mansection{strstr}
\begin{mandesc}
  \short{strstr}{locate a substring in a string matrix}
\end{mandesc}
% -- Calling sequence section
\begin{calling_sequence}
\begin{verbatim}
B=strstr(A,str)
\end{verbatim}
\end{calling_sequence}

% -- Parameters
\begin{parameters}
  \begin{varlist}
    \vname{A} : a string matrix.
    \vname{str} : a string.
    \vname{B} : a scalar matrix of same size as \verb+A+.
  \end{varlist}
\end{parameters}

\begin{mandescription}
  returns in \verb+B(i,j)+, the position of the first occurence 
  of string \verb+str+ in string \verb+A(i,j)+ or zero if \verb+str+ 
  is not a substring of \verb+A(i,j)+.
\end{mandescription}
