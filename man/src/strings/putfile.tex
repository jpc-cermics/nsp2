\mansection{putfile}
\begin{mandesc}
\short{putfile}{write a string matrix in a file in column order mode}
\end{mandesc}
% -- Calling sequence section
\begin{calling_sequence}
\begin{verbatim}
putfile(fname,S)
\end{verbatim}
\end{calling_sequence}
% -- Parameters
\begin{parameters}
  \begin{varlist}
    \vname{fname}: the file name to create given as a string.
    \vname{S}: a string matrix giving the line contents of the file to be created. The 
    string matrix is accessed in column order mode.
  \end{varlist}
\end{parameters}

\begin{mandescription}
  Writes in file \verb+fname+ the contents of \verb+S+ in column order (i.e like \verb+S(:)+)
\end{mandescription}

\begin{examples}
  \begin{mintednsp}{nsp} 
    A = string([1,2;3,4]);
    putfile('A.txt',A);
  \end{mintednsp}
\end{examples}
