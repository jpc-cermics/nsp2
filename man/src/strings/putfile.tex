\mansection{putfile}
\begin{mandesc}
\short{putfile}{write a string matrix in a file in column order mode}
\end{mandesc}
% -- Calling sequence section
\begin{calling_sequence}
\begin{verbatim}
putfile(fname,S)
\end{verbatim}
\end{calling_sequence}
% -- Parameters
\begin{parameters}
  \begin{varlist}
    \vname{fname} : the file name to create given as a string.
    \vname{A} : a string matrix.
  \end{varlist}
\end{parameters}

\begin{mandescription}
writes in file \verb+fname+ the contents of \verb+A+ in column order (i.e like \verb+A(:)+)
\end{mandescription}

\begin{examples}
  \begin{program}
    \HCode{A = string([1,2;3,4]);\Hnewline
    putfile('A.txt',A);\Hnewline
    }
  \end{program}
\end{examples}
