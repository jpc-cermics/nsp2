% -*- mode: latex -*-

\mansection{split}
\begin{mandesc}
  \short{split}{string splitting}
\end{mandesc}
% -- Calling sequence section
\begin{calling_sequence}
\begin{verbatim}
B=split(str [, msep=bool , sep= string]) 
\end{verbatim}
\end{calling_sequence}

% -- Parameters
\begin{parameters}
  \begin{varlist}
    \vname{str} : a string.
    \vname{B} : a string row vector.
  \end{varlist}
\end{parameters}

\begin{mandescription}
splits the given string using string as a set of one char separator 
(default value is \verb+' \t\n'+). If \verb+msep=%t+ then consecutive char
separator are accepted as one valid separator. 
\end{mandescription}

\begin{examples}
  \begin{program}
    \HCode{split('la terre est bleue');\Hnewline
    split('la terre est   bleue',msep=\%t);\Hnewline
    split('la terre,est; bleue',msep=\%t,sep=',; ');}
  \end{program}
\end{examples}
