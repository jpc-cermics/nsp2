% -*- mode: latex -*-

\mansection{ascii}
\begin{mandesc}
  \short{ascii}{ascii conversion}
\end{mandesc}

% -- Calling sequence section
\begin{calling_sequence}
\begin{verbatim}
A=ascii(str)  
str=ascii(A)  
\end{verbatim}
\end{calling_sequence}

% -- Parameters
\begin{parameters}
  \begin{varlist}
    \vname{A}: row vector with ascii codes
    \vname{str}: string
  \end{varlist}
\end{parameters}
\begin{mandescription}
\begin{itemize}
  \item \verb+ascii(str)+ returns in a row vector the ascii codes 
    of the string \verb+str+. 
  \item \verb+ascii(A)+ returns the string obtained from the contents 
    of the row vector \verb+A+. If \verb+A+ is a matrix the codes are read
    row by row to build the resulting string.
\end{itemize}


\end{mandescription}
