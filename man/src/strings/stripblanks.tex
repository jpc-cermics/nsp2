\mansection{stripblanks}
\begin{mandesc}
  \short{stripblanks}{removes leading and trailing white-space characters}
\end{mandesc}
% -- Calling sequence section
\begin{calling_sequence}
\begin{verbatim}
B=stripblanks(A)
\end{verbatim}
\end{calling_sequence}

% -- Parameters
\begin{parameters}
  \begin{varlist}
    \vname{B,A}: string matrices.
  \end{varlist}
\end{parameters}

\begin{mandescription}
removes leading and trailing white-space characters, i.e characters for which 
\verb+isspace+ returns \verb+%t+.
\end{mandescription}
